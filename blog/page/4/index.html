var email = require('mailer');

email.send({
      host : "smtp.gmail.com",
      port : "25",
      domain : "smtp.gmail.com",
      to : "jiangrongyong@gmail.com",
      from : "jiangrongyong@gmail.com",
      subject : "小米手机2竞购",
      body: "小米32G*2",
      authentication : "login",
      username : "jiangrongyong",
      password : "***"
}, function(err, result){
      if(err){ console.log(err); }
      console.log(new Date() + 'sended.');
});

start.sh

#!/bin/bash

export PATH=/usr/lib64/qt-3.3/bin:/usr/local/sbin:/usr/local/bin:/sbin:/bin:/usr/sbin:/usr/bin:/root/bin:$PATH

mails[0]='mailer.js'
repeat=3

for((i=1;i<=$repeat;i++));do
    for item in ${mails[@]}; do
        node /data/app/node-mailer-test/$item &
    done
    sleep 1
done

Nginx配置

配置多子站点时，例如游戏各种子官网，可以使用通配符或正则表达式的方式配置

server {
    listen       80;
    server_name  *.wan.xoyo.com;
    root   /Users/jiangrongyong/Program/online_code/official_website/guanwang;
    index  index.html index.htm index.php;
    location ~ \.php$ {
        fastcgi_pass   127.0.0.1:9000;
        fastcgi_index  index.php;
        fastcgi_param  SCRIPT_FILENAME $document_root$fastcgi_script_name;
        include        fastcgi_params;
    }
}

需要注意的是，PHP在获取$_SERVER["SERVER_NAME"]时，会获取到*.wan.xoyo.com，如果需要获取真实的域名，修改fastcgi_params，把SERVER_NAME的值改成host

fastcgi_param  SERVER_NAME        $host;

暗黑破坏神3

今天在mac上体验了一把暗黑3，把感受mark一下

1. 选的野蛮人，打击感很强烈，音效逼真，虽然前期没几个技能，但是就用普通的打击也非常的爽。
2. 游戏在13寸高配MacBookPro下总体流畅，除了剧情动画和部分地点会卡。
3. 客户端下载策略，分成安装、可执行、最佳体验三个阶段，不要求用户必须下载完整客户端也可以先体验游戏，并且语言包也是按需下载。
4. 虽然是传统单机的一次性付费模式，但是提供试玩，并且是一个完整的剧情，非常厚道。
5. 游戏分美、欧、亚三个服务器，刚进游戏时找不到选服的地方，找了好一会才发现在设置里面，跟常规体验有出入。
6. 游戏必须使用鼠标操作，无法用键盘wasd操作，对于键盘控来说很累人。
7. 当接到一个任务时，npc会把任务剧情完整读一遍，非常赞，不再需要看累人的文字剧情了。
8. 亚洲服务器没有拍卖。
